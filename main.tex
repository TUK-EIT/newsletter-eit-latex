%% Specfic Newsletter Template Definitions
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Define separators
\newcommand{\HorRule}[1]{\noindent\rule{\linewidth}{#1}} % Creating a horizontal rule
\newcommand{\SepRule}{\noindent							 % Creating a separator
						\begin{center}
							\rule{250pt}{1pt}
						\end{center}
						}						

% Define Title en News input
\newcommand{\JournalName}[1]{%
	\begin{center}
		\Huge \ptc \color{tuk-blue}
%		\usefont{T1}{augie}{m}{n}
		#1%
	\end{center}	
	\par
	\normalsize
	\normalfont
}

\newcommand{\JournalIssue}[1]{%
	\begin{center}
		\Large \ptc \color{tuk-coldgrey}
%		\usefont{T1}{augie}{m}{n}
		#1%
	\end{center}	
	\par
	\normalsize
	\normalfont
}
		
\newcommand{\NewsItem}[1]{%
%	\usefont{T1}{augie}{m}{n}
	\noindent
	\ptc
	\color{tuk-blue}
	\large #1 \vspace{4pt}
	\par \normalsize \normalfont \normalcolor
}
		
\newcommand{\NewsAuthor}[1]{%
	\pt \small
	\color{tuk-coldgrey}
	\hfill by \textsc{#1} \vspace{4pt}
	\par \normalsize \normalfont \normalcolor
}

   {\end{verse} \end{minipage}}
